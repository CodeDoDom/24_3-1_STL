		exit(0);
	}

	// 저장할 파일을 덧붙여 쓰기모드로 연다.
	std::ofstream out{ "2024 1학기 STL 화56목56 강의저장.txt", std::ios::app };

	//for (char c : v)
	//	std::cout << c;

	out << std::endl << std::endl;
	std::copy(v.begin(), v.end(), std::ostreambuf_iterator<char>{out});
}
